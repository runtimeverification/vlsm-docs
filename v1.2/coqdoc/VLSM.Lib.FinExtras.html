Finite type utility definitions and lemmas


Fixpoint up_to_n_listing
  (n : nat)
  : list nat
  :=
  match n with
  | 0 => []
  | S n' => n' :: up_to_n_listing n'
  end.

Lemma up_to_n_listing_length
  (n : nat)
  : length (up_to_n_listing n) = n.
Proof.
  induction n; simpl; congruence.
Qed.

Proof.

Qed.
